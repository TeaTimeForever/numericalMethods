\subsection{Подготовка: создание уравнения, вспомогательные элементы}

\begin{displaymath}
  \begin{array}{ccc}
    f(x) = x / sin^2(3x)  \\
    f'(x) = (1 - 6x cos(3 x) / sin(3 x)) 1/sin^2(3 x)\\
    f''(x) = 6/sin^4(3x) * (6x - sin(6x) + 3x * cos(6x))
  \end{array}
\end{displaymath}

\begin{displaymath}
  \begin{array}{ccc}
    f(x) = N_{g}  \\
    x / sin^2(3x) = 17 \\
    x / sin^2(3x) - 17 = 0 \\
  \end{array}
\end{displaymath}
